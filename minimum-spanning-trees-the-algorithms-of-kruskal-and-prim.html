Kruskal's algorithm

Kruskal's algorithm finds a safe edge to add to the growing forest by finding, of all the edges that connect any two trees in the forest, an edge $(u,v)$ of least weight. It uses a disjoint-set data structure to maintain several disjoint sets of elements.

@inlinable
public func kruskal(G : AdjacencyList) -> [(Int, Int, Int)] {
var ufs = UnionFindSet<Int>()
for v in 0 ..< G.count {
ufs.make_set(v)
}
var edges = [(Int, Int, Int)]()
for e in G.edges.sorted(by: { 736039E6-B7F5-47CD-BEBA-94F0A0FC45F71.2 }) {
let u = e.0
let v = e.1
if ufs.find_set(u) != ufs.find_set(v) {
edges.append(e)
ufs.union(u, v)
}
}
return edges
}
